Лабораторна робота 7

Мета роботи

Опанувати теоретичні основи обробки рядків та текстових функціонального програмування та розробити програми їх реалізації

Вибір мови програмування та IDE

Для виконання даної лабораторної роботи було використано мову функціонального програмування Scheme та IDE DrRacket. Мова Scheme надає великий набір числових типів і операцій з ними, має неявні типи даних та статичну область видимості. DrRacket дозволяє програмувати на великій кількості мов функціонального програмування та має зручний інтерфейс.

Завдання

Умова завдання

Записати в текстовий файл n рядків тексту, що задаються програмою на функціональній мові програмування. Зчитати рядки із створеного програмою файлу, вивести їх на екран. В початок кожного рядка тексту вставити задане користувачем слово. Записати в новий текстовий файл результат обробки тексту.

Код програми:

                
(define (write-to-file n)
    (let loop ((i 1)
        (port (open-output-file "/home/alex/first_func_prog.txt")))
            (cond ((> i n) (close-output-port port))
                  (else
                        (display "Введіть ")
                        (display i)
                        (display "-ий рядок: ")
                        (write (symbol->string(read)) port)
                        (newline port)
                        (loop (+ i 1) port)
                  )
            )
    )
)
                  
(define (read-from-file name)
    (let ((port (open-input-file name)))
        (read-from-file-helper port)
        (close-input-port port)
        (newline)))
                  
(define (read-from-file-helper port)
    (let ((action (read port)))
        (if (eof-object? action)
            (newline)
            (begin (write action)
                   (newline)
                   (read-from-file-helper port)))))
                  
(define (write-concatenated-strings conc-str file-name)
    (let ((port-read (open-input-file file-name)) (port-write (open-output-file "/home/alex/second_func_prog.txt")))
        (write-concatenated-strings-helper conc-str port-read port-write)
        (close-input-port port-read)
        (close-output-port port-write)
        (newline))
)
                 
(define (write-concatenated-strings-helper conc-str port-read port-write)
    (let ((action (read port-read)))
        (if (eof-object? action)
            (newline)
            (begin (write (string-append conc-str action) port-write)
                (newline port-write)
                (write-concatenated-strings-helper conc-str port-read port-write))))
)
                  
(display "Введіть значення n: ")
(define n (read))
                 
(write-to-file n)
                 
(read-from-file "/home/alex/first_func_prog.txt")
                  
(display "Введіть слово, яке треба вставити в кожен рядок: ")
(define adding-string (symbol->string(read)))
                 
(write-concatenated-strings adding-string "/home/alex/first_func_prog.txt")
